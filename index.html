        System.out.println(str1 + " and " + str2 + " are not anagram.");
      }
    }
    else {
      System.out.println(str1 + " and " + str2 + " are not anagram.");
    }
    input.close();
  }
}

PALINDROME

import java.util.Scanner;
class Main {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    System.out.print("Enter a string to check if it's a palindrome: ");
    String str = scanner.nextLine();
    String reverseStr = "";
    int strLength = str.length();
    for (int i = strLength - 1; i >= 0; --i) {
      reverseStr += str.charAt(i);
    }
    if (str.toLowerCase().equals(reverseStr.toLowerCase())) {
      System.out.println(str + " is a Palindrome String.");
    } else {
      System.out.println(str + " is not a Palindrome String.");
    }
    scanner.close();
  }
}

import java.nio.file.Paths;

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the file path: ");
        String filePath = scanner.nextLine();
        System.out.print("Enter the string to replace: ");
        String oldString = scanner.nextLine();
        System.out.print("Enter the replacement string: ");
        String newString = scanner.nextLine();
        try {
            String fileContent = new String(Files.readAllBytes(Paths.get(filePath)));
            String modifiedContent = fileContent.replace(oldString, newString);

            fileWriter.write(modifiedContent);
            fileWriter.close();
            System.out.println("String replaced successfully.");
        } catch (IOException e) {
            System.out.println("Error occurred while accessing the file: " + e.getMessage());
        } finally {
            scanner.close();
        }
    }
}

STACK PUSH SEARCH

import java.util.Scanner;
class Stack {
    private static final int MAX_SIZE = 30;
    private int[] arr;
    private int top;
    public Stack() {
        arr = new int[MAX_SIZE];
        top = -1;
    }
    public boolean isEmpty() {
        return top == -1;
    }
    public boolean isFull() {
        return top == MAX_SIZE - 1;
    }
    public void push(int value) {
        if (isFull()) {
            System.out.println("Stack overflow. Unable to push element " + value);
        } else {
            arr[++top] = value;
        }
    }
    public boolean search(int value) {
        if (isEmpty()) {
            System.out.println("Stack is empty. Cannot search.");
            return false;
        }
        for (int i = 0; i <= top; i++) {
            if (arr[i] == value) {
                System.out.println("Element " + value + " found at position " + i);
                return true;
            }
        }
        System.out.println("Element " + value + " not found in the stack.");
        return false;
    }
}
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the number of elements to push onto the stack: ");
        int n = scanner.nextInt();
        Stack stack = new Stack();
        System.out.println("Enter " + n + " elements to push onto the stack:");
        for (int i = 0; i < n; i++) {
            int element = scanner.nextInt();
            stack.push(element);
        }
        System.out.print("Enter an element to search in the stack: ");
        int searchElement = scanner.nextInt();
        stack.search(searchElement);
        scanner.close();
    }
}

        Set< Character > vowels = new HashSet<>();
        vowels.add('a');
        vowels.add('e');
        vowels.add('i');
        vowels.add('o');
        vowels.add('u');
        vowels.add('A');
        vowels.add('E');
        vowels.add('I');
        vowels.add('O');
        vowels.add('U');
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < input.length(); i++) {
            char c = input.charAt(i);
            if (!vowels.contains(c)) {
                sb.append(c);
            }
        }
        return sb.toString();
    }
    public static void main(String[] args) {
        if (args.length < 1) {
            System.out.println("Please provide the file path as an argument.");
            return;
        }
        String filePath = args[0]; 
        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = br.readLine()) != null) {
                System.out.println(removeVowels(line));
            }
        } catch (IOException e) {
            System.err.println("Error reading the file: " + e.getMessage());
        }
    }
}

MEDIAN FINDER

import java.util.Arrays;
import java.util.Scanner;
public class CTJ17190 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Please enter a series of float numbers separated by spaces:");
        String inputLine = scanner.nextLine();
        String[] inputValues = inputLine.split(" ");
        float[] floatArray = new float[inputValues.length];
        for (int i = 0; i < inputValues.length; i++) {
            floatArray[i] = Float.parseFloat(inputValues[i]);
        }
        Arrays.sort(floatArray);
        float median;
        int length = floatArray.length;
        if (length % 2 == 0) {
            median = (floatArray[length / 2 - 1] + floatArray[length / 2]) / 2.0f;
        } else {
            median = floatArray[length / 2];
        }
        System.out.println("The median is: " + median);
        scanner.close();
    }
}

PASSWORD CHECKER

import java.util.Scanner;
public class PasswordChecker {
    public static boolean isStrongPassword(String password) {
        // Define password rules
        int minLength = 8; // Minimum length for a strong password
        boolean hasUpperCase = false;
        boolean hasLowerCase = false;
        boolean hasDigit = false;
        boolean hasSpecialChar = false;
        if (password.length() < minLength) {
            return false; // Too short
        }
        for (int i = 0; i < password.length(); i++) {
            char c = password.charAt(i);
            if (Character.isUpperCase(c)) {
                hasUpperCase = true;
            } else if (Character.isLowerCase(c)) {
                hasLowerCase = true;
            } else if (Character.isDigit(c)) {
                hasDigit = true;
            } else if (isSpecialCharacter(c)) {
                hasSpecialChar = true;
            }
            if (hasUpperCase && hasLowerCase && hasDigit && hasSpecialChar) {
                return true;
            }
        }
        return hasUpperCase && hasLowerCase && hasDigit && hasSpecialChar;
    }
    public static boolean isSpecialCharacter(char c) {
        String specialCharacters = "!@#$%^&*()-_=+[]{}|;:',.<>?/`~";
        return specialCharacters.indexOf(c) != -1;
    }
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a password to check its strength:");
        String password = scanner.nextLine();
        int passwordLength = password.length();
        boolean isStrong = isStrongPassword(password);
        if (isStrong) {
            System.out.println("The password is strong.");
        } else {
            System.out.println("The password is weak. A strong password should have:");
            System.out.println("- At least 8 characters");
            System.out.println("- Both upper and lower case letters");
            System.out.println("- At least one digit");
            System.out.println("- At least one special character");
        }
        System.out.println("Length of the password: " + passwordLength);
        scanner.close();
    }
}

ANAGRAM GENERATOR

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;
public class AnagramGenerator {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a word: ");
        String word = scanner.nextLine();
        List< String> anagrams = generateAnagrams(word);
        System.out.println("Anagrams:");
        for (String anagram : anagrams) {
            System.out.println(anagram);
        }
        scanner.close();
    }
    private static List< String> generateAnagrams(String word) {
        List< String> result = new ArrayList<>();
        permute("", word, result);
        return result;
    }
    private static void permute(String prefix, String str, List< String> result) {
        int n = str.length();
        if (n == 0) {
            result.add(prefix);
        } else {
            for (int i = 0; i < n; i++) {
                permute(prefix + str.charAt(i), str.substring(0, i) + str.substring(i + 1, n), result);
            }
        }
    }
}

import java.util.Scanner;
public class MinMaxFinder {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter integers one by one (enter a non-integer to stop):");
        boolean firstInput = true;
        int max = Integer.MIN_VALUE;
        int min = Integer.MAX_VALUE;
        while (scanner.hasNextInt()) {
            int number = scanner.nextInt();
            if (firstInput) {
                max = number;
                min = number;
                firstInput = false;
            } else {
                if (number > max) {
                    max = number;
                }
                if (number < min) {
                    min = number;
                }
            }
        }
        System.out.println("Maximum: " + max);
        System.out.println("Minimum: " + min);
        scanner.close();
    }
}

import java.util.Scanner;
import java.util.TreeSet;
public class TreeSetThreshold {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter elements for the TreeSet, separated by spaces: ");
        String input = scanner.nextLine();
        TreeSet< Integer> treeSet = new TreeSet<>();
        String[] elements = input.split(" ");
        for (String element : elements) {
            try {
                int value = Integer.parseInt(element);
                treeSet.add(value);
            } catch (NumberFormatException e) {
                System.out.println("Invalid input. Please enter integers only.");
                scanner.close();
                return;
            }
        }
        System.out.print("Enter the threshold value: ");
        int threshold = scanner.nextInt();
        treeSet.headSet(threshold + 1).clear();
        if (treeSet.isEmpty()) {
            System.out.println("No greater elements");
        } else {
            System.out.print("TreeSet after filtering: ");
            for (Integer element : treeSet) {
                System.out.print(element + " ");
            }
            System.out.println();
        }
        scanner.close();
    }
}
